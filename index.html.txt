<script>
let currentTab = 'text';
function switchTab(tab) {
  currentTab = tab;
  document.querySelectorAll('.tab-content').forEach(tc => tc.style.display='none');
  document.getElementById(tab+'-tab').style.display='block';
  document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));
  document.querySelector(`.tab-btn[onclick="switchTab('${tab}')"]`).classList.add('active');
}

document.getElementById('generateBtn').addEventListener('click', async ()=>{
  const videoPlayer = document.getElementById('videoPlayer');
  const duration = document.getElementById('duration').value;
  const ratio = document.getElementById('ratio').value;

  if(currentTab==='text'){
    const text = document.getElementById('textInput').value;
    if(!text) return alert('Enter text!');
    console.log(`Text-to-Video: "${text}", Duration: ${duration}s, Ratio: ${ratio}`);
  } else if(currentTab==='image'){
    const img = document.getElementById('imageInput').files[0];
    if(!img) return alert('Upload an image!');
    console.log(`Image-to-Video: "${img.name}", Duration: ${duration}s, Ratio: ${ratio}`);
  } else if(currentTab==='voice'){
    const voice = document.getElementById('voiceInput').files[0];
    if(!voice) return alert('Upload a voice file!');
    console.log(`Voice-to-Video: "${voice.name}", Duration: ${duration}s, Ratio: ${ratio}`);
  }

  // Placeholder video
  videoPlayer.src = "https://sample-videos.com/video123/mp4/720/big_buck_bunny_720p_1mb.mp4";
  videoPlayer.play();
});
</script>
